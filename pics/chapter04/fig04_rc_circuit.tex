% Circuit layout:
%   Bottom-left (0,0) -- bottom-right (6,0): return wire
%   Left side (0,0) to (0,4): voltage source U_0
%   Top-left (0,4): switch S (open position)
%   Top wire with resistor R from switch to top-right corner
%   Capacitor C on the right side (6,4) to (6,0)

% Voltage source on the left side (bottom to top)
\draw (0,0) to[battery1, v_=$U_0$, color=my_darkblue] (0,4);

% Wire from voltage source to switch pivot
\draw[my_darkblue, thick] (0,4) -- (1.0,4);

% Pivot dot
\filldraw[my_darkblue] (1.0,4) circle (2pt);

% Tilted switch blade (open position: rotated ~25 degrees upward)
\draw[my_darkblue, thick] (1.0,4) -- (1.85,4.4);

% Fixed contact dot
\filldraw[my_darkblue] (2.0,4) circle (2pt);

% Wire from fixed contact to resistor
\draw[my_darkblue, thick] (2.0,4) -- (2.7,4);

% Label S above the switch blade
\node[above, font=\small, my_darkgray] at (1.5,4.45) {$S$};

% Resistor on the top wire
\draw (2.7,4) to[R, l=$R$, color=my_darkblue] (5.5,4);

% Top-right corner wire to capacitor
\draw[my_darkblue] (5.5,4) -- (6,4);

% Capacitor on the right side (top to bottom), voltage label U_C
\draw (6,4) to[C, l_=$C$, v^=$U_C$, color=my_darkblue] (6,0);

% Bottom return wire
\draw[my_darkblue] (6,0) -- (0,0);
